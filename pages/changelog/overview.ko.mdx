# 데스크톱 변경 로그

우리는 이 로그를 사용하여 우리가 완료한 작업을 모두 기록합니다. 이것은 우리가 얼마나 많은 진전을 이루었는지 볼 수 있게 도와줍니다. 이 변경 로그는 <Link href='https://useanything.com/download'><u>TeamplGPT 데스크톱 앱</u></Link>의 변경 사항만 추적합니다.

아래 카드들을 클릭하여 최근 변경 로그를 읽을 수 있습니다:

<Cards>
  {["v1.5.8", "v1.5.7", "v1.5.6", "v1.5.5"].map(function (tag) {
    return (
      <Card title={`변경 로그 ${tag}`} href={tag}>
        <Image
          src="/images/product/changelog/header-image.png"
          height={1080}
          width={1920}
          quality={100}
          alt={`TeamplGPT 데스크톱 변경 로그 ${tag}`}
        />
      </Card>
    );
  })}
</Cards>

export const Card = Object.assign(
  // 카드 컴포넌트를 복사하고 기본 속성을 추가
  Cards.Card.bind(),
  {
    displayName: "Card",
    defaultProps: {
      image: true,
      arrow: true,
      target: "_self",
    },
  }
);
